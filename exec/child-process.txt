const { exec } = require('child_process');
exec('PowerShell.exe Get-EventLog -Newest 5 -LogName "Application"', (error, stdout, stderr) => {
  if (error) {
    console.error(`exec error: ${error}`);
    return;
  }
  console.log(`stdout: ${stdout}`);
  console.log(`stderr: ${stderr}`);
});
